function form_start($return = 0, $action = "?", $method = "GET", $extra = "")

Examples:
<?php
//simple call
$code->form_start();
// we can use the parameters like this
$code->form_start(0,"login.php","POST","form1");
// but we can also do this
$code->form_start(0,"login.php","POST",array("class" => "form2", "id" => "form_1"));
?>
Output:
<form action="?" method="GET">
<form action="login.php" method="POST" class="form1">
<form action="login.php" method="POST" class="form2" id="form_1">
function form_input($return = 0, $type="text", $name='input1', $extra = "", $endform = 0)

</form>

function js($return = 0,$file = '',$extra = '',$code = '')

Examples:
<?php
// simple call
$code->js();
// we can use the arguments like this

?>

<script src="/folder/script.js" type="text/javascript"></script>

<script type="text/javascript">document.write("Hello !")</script>

<script src="/folder/script.js" type="text/javascript" defer="1">document.write("Hello !")</script>
